⚙️ Cómo conectar tu Google Sheet
  1. Abre tu Google Sheet → Extensiones → Apps Script
  2. Selecciona todo (Ctrl+A), pega el código de abajo
  3. Guarda (Ctrl+S) → Implementar → Nueva implementación
  4. Tipo: Aplicación web · Acceso: Cualquier usuario
  5. Copia la URL /exec y pégala aquí abajo
Código para Apps Script 
function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data  = JSON.parse(e.postData.contents);
    var row   = data.row;
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(data.headers);
    }
    sheet.appendRow(row);
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok' }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({ status: 'ok', message: 'Escaner Tec activo' }))
    .setMimeType(ContentService.MimeType.JSON);
}
