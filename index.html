⚙️ Cómo configurar tu Google Sheet
  1. Abre tu Google Sheet y ve a Extensiones → Apps Script
  2. Borra el contenido y pega el código de abajo
  3. Guarda (💾) y haz clic en Implementar → Nueva implementación
  4. Tipo: Aplicación web · Acceso: Cualquier usuario
  5. Copia la URL generada y pégala aquí abajo
Código para Apps Script 
function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data  = JSON.parse(e.postData.contents);
    var row   = data.row; // array of values in column order

    // Write headers on first row if empty
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(data.headers);
    }
    sheet.appendRow(row);

    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok' }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
